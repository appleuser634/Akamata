Akamata

A modern Zig 0.16 web framework
One source — runs on a VPS, runs on Cloudflare Workers
v0.3 · 2026

Why we built it

  • Bring Zig's performance + safety (zero deps, explicit memory) to the web
  • Ship "a small static binary for VPS" AND "a Cloudflare Workers wasm" from one codebase
  • Hono-style DX (handlers, middleware, typed JSON) in Zig
  • Transparent DB layer: SQLite / Turso / D1 swappable by URL

Result

Production-quality · 53M requests with zero leak · P99 88µs · audited HTTP smuggling defense

Hello, Akamata

If you've used Express or Hono, you already know how to read this.

const std = @import("std");
const am  = @import("akamata");

const State = struct {};

pub fn main() !void {
    var gpa: std.heap.DebugAllocator(.{}) = .init;
    defer _ = gpa.deinit();

    var app = am.App(State).init(gpa.allocator(), .{});
    defer app.deinit();

    _ = try app.useAll(am.mw.recover(State));
    _ = try app.get("/", hello);

    try app.serve(.{ .port = 8080 });
}

fn hello(c: *am.Context(State)) !void {
    try c.text("Hello, Akamata!");
}

The Model layer — one struct does it all

pub const Todo = struct {
    id: ?i64 = null,
    title: []const u8,
    priority: i32 = 3,
    completed: bool = false,
    created_at: ?i64 = null,

    pub const __schema = .{
        .table = "todos",
        .indexes = .{ .{ .{ "completed", "priority" }, .index } },
        .defaults  = .{ .created_at = "unixepoch()" },
        .validates = .{
            .title    = .{ am.model.rule.required, am.model.rule.max_len(200) },
            .priority = .{ am.model.rule.range(1, 3) },
        },
    };
};

const Todos = am.model.repo(Todo);

Validation, defaults, indexes, DDL, migrations — all derived from this one declaration

Typed CRUD — no raw SQL needed

// reads
const one    = try Todos.find(db, arena, 42);          // ?Todo
const all    = try Todos.all(db, arena);              // []Todo
const highs  = try Todos.where(db, arena, .{ .priority = 1 });

// writes
const created = try Todos.create(db, arena, .{ .title = "buy milk" });
try Todos.save(db, arena, &updated);
try Todos.delete(db, 7);

// escape hatch — arbitrary SQL still maps to T
const rows = try Todos.queryRaw(db, arena,
    "SELECT * FROM todos WHERE created_at > ? ORDER BY id", .{cutoff});

// kills N+1 — one IN query fetches all children

Handlers, abbreviated

Before — 13 lines

fn createTodo(c: *Ctx) !void {
    const Body = struct { title: []const u8 };
    const body = c.req.json(Body) catch {
        return c.json(.{
            .error_kind = "bad_request",
            .message = "expected {title}",
        }, 400);
    };
    const errs = try Todos.validate(body, c.arena);
    if (errs.len > 0) {
        return c.json(.{ .errors = errs }, 422);
    }
    const t = try Todos.create(c.db(), c.arena, body);
    try c.json(t, 201);
}

After — 3 lines

fn createTodo(c: *Ctx) !void {
    const todo = (try c.input(Todo)) orelse return;
    const t = try Todos.create(c.db(), c.arena, todo);
    try c.json(t, 201);
}
  • c.input(T) parses + validates
  • On failure 400 / 422 auto-emitted
  • orelse return for early exit

Auto-migration

At startup, diff the model's __schema against the live DB, generate only the needed ALTERs.

const plan = try am.model.migrate.diff(arena, db, &all_models);
try am.model.migrate.apply(arena, db, plan);
// → CREATE TABLE / ADD COLUMN / CREATE INDEX as needed

Or versioned files

$ akamata migrate generate add_email
→ creates migrations/20260523120000_add_email.sql

$ ./myapp migrate-up
→ checks schema_migrations, applies pending ones

Both modes work on SQLite / Turso / D1 alike. On Workers it goes through JSPI.

One command deploy

$ akamata deploy --workers \
    --config=deploy/wrangler.toml \
    --migrate=<(./zig-out/bin/myapp --print-schema)
  1. Reads [[d1_databases]] from wrangler.toml
  2. Placeholder UUID? Runs wrangler d1 create, writes the real one back
  3. DB already exists? Picks it up via wrangler d1 list
  4. Applies the schema you piped through --migrate
  5. zig build -Dbackend=workers
  6. wrangler deploy

Idempotent on re-runs. Safe to call from CI.

The magic: JSPI

Lets Zig call D1's async API as if it were synchronous.

JS host (index.mjs)

d1_step: new WebAssembly.Suspending(
  async (h) => {
    const e = d1stmts.get(h);
    return await e.iter.next();
  }
);

handleFetchAsync = WebAssembly.promising(
  exports.handle_fetch);

Zig side (d1.zig)

extern "akamata_d1"
    fn d1_step(stmt: i32) i32;

// looks synchronous
while ((try stmt.step()) == .row)
    try rows.append(arena, row);

V8 parks the wasm stack across awaits and resumes it. Zero Zig changes, zero binary bloat vs. Asyncify.

Reliability — production audited

HTTP smuggling defense

CL + TE rejected, duplicate CL rejected, malformed chunked rejected. 902 adversarial fuzz attempts, zero panics.

JWT done right

Stable under load

TLS cert verification

Graceful shutdown

OOM safe

WebSocket — upgrade from any route

_ = try app.ws("/rooms/:id/ws", wsRoom);

fn wsRoom(c: *Ctx) !void {
    const room_id = try c.req.paramAs(u64, "id");
    var conn = try am.ws.upgrade(Ctx, c, .{ .max_message_bytes = 64 * 1024 });
    try c.state().hub.attach(room_id, &conn);
    defer c.state().hub.detach(room_id, &conn);

    while (true) {
        const msg = conn.readMessage() catch return;
        defer msg.deinit();
        if (msg.opcode == .text)
            try c.state().hub.broadcast(room_id, msg.payload);
    }
}

On Workers, WS sessions are delegated to a Durable Object automatically — your handler code doesn't change.

Validation — declare it, done

pub const __schema = .{
    .validates = .{
        .email = .{
            am.model.rule.required,
            am.model.rule.max_len(255),
            am.model.rule.format(.email),
        },
        .age   = .{ am.model.rule.range(0, 150) },
        .name  = .{ am.model.rule.required, am.model.rule.custom(noScriptTag) },
    },
};

fn noScriptTag(value: []const u8, _: std.mem.Allocator) ?[]const u8 {
    if (std.mem.indexOf(u8, value, "<script>") != null)
        return "must not contain <script>";
    return null;
}

Built-ins: required, min_len, max_len, range, format(.email/.url/.alphanumeric), custom, customInt

Eager loading — defeat N+1

// User has_many :posts
const users = try Users.all(db, arena);              // 1 query
const loaded = try am.model.preload.hasMany(
    User, "posts", users, db, arena);                // 1 IN query

for (loaded) |row| {
    // row.parent: User
    // row.related: []Post   ← already loaded
    std.log.info("{s} has {d} posts", .{
        row.parent.name, row.related.len
    });
}

N parents → 2 queries total. Children fetched in batch via WHERE user_id IN (?, ?, ?, ...).

Ship the HTML UI in the binary

const index_html = @embedFile("index.html");

fn index(c: *Ctx) !void {
    const accept = c.req.header("accept") orelse "";
    if (std.mem.indexOf(u8, accept, "text/html") != null) {
        return c.html(index_html);  // browsers → HTML UI
    }
    try c.json(.{ .endpoints = .{ ... } }, 200);  // curl → JSON
}
  • @embedFile bakes the file into the binary/wasm at compile time
  • Single artifact serves everything — no extra deploy step
  • Works without a separate CDN on Workers too
